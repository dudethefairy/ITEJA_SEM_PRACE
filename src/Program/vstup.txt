program mocnina;
const cislo:int=5;
var jmeno:string='pepa',exp:int,i:int, vysledek:double;

begin
pcall writeln('Me jmeno je '+jmeno+' a umocnime si cislo 5, zadejte exponent:');
pcall readln(exp);
set vysledek := cislo;

for i:= 0 to exp do begin
if i!=0 then
set vysledek := vysledek*cislo;
end;

pcall writeln('Vysledek je');
set vysledek:=vysledek/cislo;
if exp<0 then set vysledek:=1/vysledek;
pcall writeln(vysledek);
end.
