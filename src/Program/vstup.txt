program bankovniucet;
var jmeno:string, prijmeni:string, vek:int, penize:double, prijem:double;

procedure nastavOsobu(name:string,surname:string,age:int,money:double,income:double);
begin
jmeno:=name;
prijmeni:=surname;
vek:=age;
penize:=money;
prijem:=income
end;
function zjistiPrijmy(pocetMesicu:int,mesicniPrijem:double):double;
begin
zjistiPrijmy:=pocetMesicu*mesicniPrijem
end;

begin
pcall nastavOsobu('Petr','Novak',35,50000.00,20000.00);
fcall zjistiPrijmy(vek,prijem)
end.
